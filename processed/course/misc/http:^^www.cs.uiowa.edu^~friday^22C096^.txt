function loopmessage(seed) {
var message = " Welcome to 22C:096! 	 Be sure to sign up for the spring semester of Topics in Computer Science, Orientation.	The course # will be changed to 22C:003 (Sections 1 and 2) " ;
var display = " " ;
var i = 1 ;
if ( seed > 180 ) {
seed -= 2 ;
var cmd = "loopmessage(" + seed + ")" ;
window.setTimeout(cmd,100) ;
}
else if ( seed <= 180 && seed > 0 ) {
for ( i=0; i<seed; i++)
display += " " ;
display += message ;
seed -= 2 ;
var cmd = "loopmessage(" + seed + ")" ;
window.status = display ;
window.setTimeout(cmd,100) ;
}
else if ( seed <= 0 ) {
if ( -seed < message.length ) {
display += message.substring( -seed, message.length ) ;
seed -= 2 ;
var cmd = "loopmessage(" + seed + ")" ;
window.status = display ;
window.setTimeout(cmd,100) ;
}
else {
window.status = " " ;
var cmd = "loopmessage(180)"
window.setTimeout(cmd,75) ;
}
}
}
loopmessage(180);
